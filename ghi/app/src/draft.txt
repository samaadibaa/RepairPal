import React, { useState, useEffect } from "react";

function SalesHistoryList() {
  const [salespersons, setSalespersons] = useState([]);
  const [selectedSalesperson, setSelectedSalesperson] = useState("");
  const [salesHistory, setSalesHistory] = useState([]);

  useEffect(() => {
    const fetchSalespersons = async () => {
      const response = await fetch("http://localhost:8090/api/salesperson/");
      if (response.ok) {
        const data = await response.json();
        setSalespersons(data.salespersons);
      }
    };

    fetchSalespersons();
  }, []);

  useEffect(() => {
    const fetchSalesHistory = async () => {
      if (selectedSalesperson) {
        const response = await fetch(
          `http://localhost:8090/api/salesperson/${selectedSalesperson}/sales`
        );
        if (response.ok) {
          const data = await response.json();
          setSalesHistory(data.sales);
        }
      } else {
        setSalesHistory([]);
      }
    };

    fetchSalesHistory();
  }, [selectedSalesperson]);

  const handleSalespersonChange = (event) => {
    setSelectedSalesperson(event.target.value);
  };

  return (
    <div>
      <h1>Sales History List</h1>
      <div>
        <label htmlFor="salesperson">Select Salesperson:</label>
        <select
          id="salesperson"
          value={selectedSalesperson}
          onChange={handleSalespersonChange}
        >
          <option value="">All Salespersons</option>
          {salespersons &&
            salespersons.map((salesperson) => (
              <option key={salesperson.id} value={salesperson.id}>
                {salesperson.name}
              </option>
            ))}
        </select>
      </div>
      <table>
        <thead>
          <tr>
            <th>Salesperson</th>
            <th>Customer</th>
            <th>Automobile VIN</th>
            <th>Price</th>
          </tr>
        </thead>
        <tbody>
          {salesHistory.map((sale) => (
            <tr key={sale.id}>
              <td>{sale.salesperson}</td>
              <td>{sale.customer}</td>
              <td>{sale.automobileVIN}</td>
              <td>{sale.price}</td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

export default SalesHistoryList;
